**1.Array of adding 10 numbers using for**

#include <iostream><br>
 
using namespace std; <br>
 
int main() <br>
{  <br>
    int i,arr[10],sum=0;  <br>
 
    cout<<"Enter 10 elements:";  <br>
    for(i=0;i<10;++i)  <br>
        cin>>arr[i];  <br>
 
    for(i=0;i<10;++i)  <br>
        sum=sum+arr[i];  <br>
    cout<<"Sum of numbers is:"<<sum;  <br>
 
    return 0;  <br>
} <br>
